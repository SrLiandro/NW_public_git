	for_loop_effect = {
		start = start
		end = end
		value = v
		
		var:global.chinese_state^v = {
			set_variable = { china_left_border = start }
			set_variable = { china_right_border = end }
			
			#set_variable = { china_integration_of_province_time = 120 }
		}
	}
}
